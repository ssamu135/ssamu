2. POO y Mixins

# Clase básica
class Persona
  attr_accessor :nombre
  
  def initialize(nombre)
    @nombre = nombre
  end
  
  def saludar
    puts "Hola, soy #{@nombre}"
  end
end

# Módulo como Mixin
module Habilidades
  def programar(lenguaje)
    puts "Programando en #{lenguaje}..."
  end
end

# Herencia y Mixin
class Desarrollador < Persona
  include Habilidades
end

# Uso
dev = Desarrollador.new("Ana")
dev.saludar
dev.programar("Ruby")

3. Ruby on Rails (ejemplo controlador)

# app/controllers/articles_controller.rb
class ArticlesController < ApplicationController
  before_action :set_article, only: [:show, :edit, :update, :destroy]
  
  # GET /articles
  def index
    @articles = Article.all
  end
  
  # GET /articles/1
  def show
  end
  
  # GET /articles/new
  def new
    @article = Article.new
  end
  
  private
    # Use callbacks to share setup between actions
    def set_article
      @article = Article.find(params[:id])
    end
    
    # Only allow trusted parameters
    def article_params
      params.require(:article).permit(:title, :content)
    end
end
