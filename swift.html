Sintaxis Básica

Tipos y estructuras

// Variables y constantes
var nombre = "Juan"  // Mutable
let apellido = "Pérez" // Inmutable

// Tipos explícitos
var edad: Int = 30
var precio: Double = 19.99

// Colecciones
var frutas = ["Manzana", "Banana"]
var persona = ["nombre": "Ana", "edad": 25]

// Enums
enum Direccion {
    case norte, sur, este, oeste
}

Control de flujo

// If-else
if edad >= 18 {
    print("Mayor de edad")
} else {
    print("Menor de edad")
}

// Switch
switch direccion {
case .norte:
    print("Hacia el norte")
case .sur:
    print("Hacia el sur")
default:
    print("Otra dirección")
}

// Bucles
for fruta in frutas {
    print(fruta)
}

// While
var i = 0
while i < 5 {
    print(i)
    i += 1
}

Ejemplos Prácticos

1. Hola Mundo y Optionals

// Hola Mundo básico
print("¡Hola, mundo!")

// Optionals
var nombre: String? = nil
nombre = "Carlos"

// Optional binding
if let nombre = nombre {
    print("Hola, \(nombre)")
} else {
    print("Nombre no proporcionado")
}

// Guard let
func saludar(nombre: String?) {
    guard let nombre = nombre else {
        print("Nombre inválido")
        return
    }
    print("Bienvenido, \(nombre)")
}

2. Protocol-Oriented Programming

// Protocolo
protocol Dibujable {
    func dibujar() -> String
}

// Extensión con implementación por defecto
extension Dibujable {
    func dibujar() -> String {
        return "Figura genérica"
    }
}

// Estructura que adopta el protocolo
struct Circulo: Dibujable {
    var radio: Double
    
    func dibujar() -> String {
        return "Círculo con radio \(radio)"
    }
}

// Uso
let figura: Dibujable = Circulo(radio: 5.0)
print(figura.dibujar())

3. Vista simple en SwiftUI

import SwiftUI

struct ContentView: View {
    @State private var contador = 0
    
    var body: some View {
        VStack {
            Text("Contador: \(contador)")
                .font(.title)
            
            Button(action: {
                self.contador += 1
            }) {
                Text("Incrementar")
                    .padding()
                    .background(Color.blue)
                    .foregroundColor(.white)
                    .cornerRadius(10)
            }
        }
        .padding()
    }
}

// Vista previa (solo funciona en Xcode)
struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView()
    }
}
