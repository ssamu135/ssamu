Características Principales

  • Orientado a objetos: Todo en Java es un objeto
  • Multiplataforma: "Write once, run anywhere" (WORA)
  • Tipado estático: Verificación de tipos en tiempo de compilación
  • Recolección de basura: Manejo automático de memoria
  • Multihilo: Soporte para programación concurrente

1. Hola Mundo

2. POO - Clase y Objeto

// Definición de clase
public class Coche {
    // Atributos
    String marca;
    String modelo;
    int año;
    
    // Constructor
    public Coche(String marca, String modelo, int año) {
        this.marca = marca;
        this.modelo = modelo;
        this.año = año;
    }
    
    // Método
    public void arrancar() {
        System.out.println("El coche " + marca + " está arrancando");
    }
}

// Uso de la clase
public class Main {
    public static void main(String[] args) {
        Coche miCoche = new Coche("Toyota", "Corolla", 2020);
        miCoche.arrancar();
    }
}

3. Colecciones

import java.util.ArrayList;

public class EjemploListas {
    public static void main(String[] args) {
        // Crear lista
        ArrayList frutas = new ArrayList<>();
        
        // Añadir elementos
        frutas.add("Manzana");
        frutas.add("Banana");
        frutas.add("Naranja");
        
        // Recorrer lista
        for(String fruta : frutas) {
            System.out.println(fruta);
        }
    }
}
