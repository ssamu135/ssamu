Ownership y Borrowing

// Ownership (semántica de movimiento)
let s1 = String::from("hola");
let s2 = s1;  // s1 ya no es válido aquí

// Borrowing (préstamos)
fn calcular_longitud(s: &String) -> usize {
    s.len()
}

let s = String::from("texto");
let len = calcular_longitud(&s);  // Referencia inmutable
println!("{} tiene {} caracteres", s, len);

// Mutable borrowing
fn modificar(s: &mut String) {
    s.push_str(", mundo!");
}

let mut s = String::from("Hola");
modificar(&mut s);

Ejemplos Prácticos

1. Hola Mundo y pattern matching

fn main() {
    println!("¡Hola, mundo!");
    
    // Pattern matching
    let numero = 3;
    
    match numero {
        1 => println!("Uno"),
        2 | 3 | 5 | 7 => println!("Número primo"),
        4..=10 => println!("Entre 4 y 10"),
        _ => println!("Otro número"),
    }
    
    // if let
    let algun_valor = Some(5);
    if let Some(5) = algun_valor {
        println!("Es un cinco!");
    }
}

2. Concurrencia segura

use std::thread;
use std::sync::{Arc, Mutex};

fn main() {
    // Datos compartidos protegidos por Mutex
    let contador = Arc::new(Mutex::new(0));
    let mut handles = vec![];

    for _ in 0..10 {
        let contador = Arc::clone(&contador);
        let handle = thread::spawn(move || {
            let mut num = contador.lock().unwrap();
            *num += 1;
        });
        handles.push(handle);
    }

    for handle in handles {
        handle.join().unwrap();
    }

    println!("Resultado: {}", *contador.lock().unwrap());
}

3. Web Server con Actix

use actix_web::{web, App, HttpResponse, HttpServer, Responder};

async fn hola() -> impl Responder {
    HttpResponse::Ok().body("Hola desde Rust!")
}

#[actix_web::main]
async fn main() -> std::io::Result<()> {
    HttpServer::new(|| {
        App::new()
            .route("/", web::get().to(hola))
    })
    .bind("127.0.0.1:8080")?
    .run()
    .await
}
