1. Hola Mundo y funciones

#include <iostream>

    std::cout << "¡Hola, " << nombre << "!" << std::endl;
}

    // Hola Mundo básico

2. POO en C++

#include <iostream>
#include <string>

// Clase base
class Figura {
public:
    virtual double area() const = 0; // Método virtual puro
    virtual ~Figura() {}             // Destructor virtual
};

// Clase derivada
class Circulo : public Figura {
private:
    double radio;
public:
    Circulo(double r) : radio(r) {}
    
    double area() const override {
        return 3.14159 * radio * radio;
    }
};

int main() {
    Circulo c(5.0);
    std::cout << "Área: " << c.area() << std::endl;
    
    // Polimorfismo
    Figura* f = new Circulo(3.0);
    std::cout << "Área polimórfica: " << f->area() << std::endl;
    delete f;
    
    return 0;
}

3. STL (Standard Template Library)

#include <iostream>
#include <vector>
#include <algorithm>

int main() {
    // Vector (array dinámico)
    std::vector<int> numeros = {5, 2, 8, 1, 9};
    
    // Iteración con range-based for (C++11)
    for (int num : numeros) {
        std::cout << num << " ";
    }
    std::cout << std::endl;
    
    // Algoritmo STL (ordenación)
    std::sort(numeros.begin(), numeros.end());
    
    // Función lambda (C++11)
    auto cuadrado = [](int x) { return x * x; };
    std::cout << "5 al cuadrado: " << cuadrado(5) << std::endl;
    
    return 0;
}
