SQL

Creado en 1974 IBM (Donald D. Chamberlin) Usado en el 90% de bases de datos

Características Principales

  • Lenguaje declarativo: Describe qué datos obtener, no cómo
  • Estandarizado: ANSI/ISO SQL (con variantes por motor)
  • CRUD completo: Create, Read, Update, Delete
  • Transacciones: ACID (Atomicidad, Consistencia, Aislamiento, Durabilidad)
  • Multiplataforma: MySQL, PostgreSQL, SQL Server, Oracle, etc.

Sintaxis Básica

DDL (Data Definition Language)

-- Crear tabla
CREATE TABLE Empleados (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(50) NOT NULL,
    salario DECIMAL(10,2) CHECK (salario > 0),
    departamento_id INT,
    fecha_contratacion DATE DEFAULT CURRENT_DATE,
    FOREIGN KEY (departamento_id) REFERENCES Departamentos(id)
);

-- Modificar tabla
ALTER TABLE Empleados ADD COLUMN email VARCHAR(100) UNIQUE;

-- Eliminar tabla
DROP TABLE Empleados;

DML (Data Manipulation Language)

-- Insertar datos
INSERT INTO Empleados (nombre, salario, departamento_id)
VALUES ('Ana López', 45000.00, 2);

-- Actualizar datos
UPDATE Empleados 
SET salario = salario * 1.10 
WHERE departamento_id = 3;

-- Eliminar datos
DELETE FROM Empleados 
WHERE fecha_contratacion < '2020-01-01';

Ejemplos Prácticos

1. Consultas básicas

-- Selección con filtros
SELECT nombre, salario 
FROM Empleados 
WHERE salario > 50000 
ORDER BY salario DESC;

-- Agregación
SELECT 
    departamento_id,
    COUNT(*) AS total_empleados,
    AVG(salario) AS salario_promedio
FROM Empleados
GROUP BY departamento_id
HAVING COUNT(*) > 5;

-- Subconsultas
SELECT nombre 
FROM Empleados 
WHERE departamento_id IN (
    SELECT id FROM Departamentos WHERE ubicacion = 'Madrid'
);

2. JOINS avanzados

-- INNER JOIN
SELECT 
    e.nombre AS empleado,
    d.nombre AS departamento,
    p.nombre AS proyecto
FROM Empleados e
INNER JOIN Departamentos d ON e.departamento_id = d.id
INNER JOIN Empleados_Proyectos ep ON e.id = ep.empleado_id
INNER JOIN Proyectos p ON ep.proyecto_id = p.id;

-- LEFT JOIN (incluye empleados sin departamento)
SELECT 
    e.nombre,
    d.nombre AS departamento
FROM Empleados e
LEFT JOIN Departamentos d ON e.departamento_id = d.id;

-- SELF JOIN (jerarquía de empleados)
SELECT 
    e1.nombre AS empleado,
    e2.nombre AS supervisor
FROM Empleados e1
LEFT JOIN Empleados e2 ON e1.supervisor_id = e2.id;

3. Transacciones y procedimientos

-- Transacción ACID
BEGIN TRANSACTION;

UPDATE Cuentas 
SET balance = balance - 1000 
WHERE id = 1;

UPDATE Cuentas 
SET balance = balance + 1000 
WHERE id = 2;

-- Si todo va bien
COMMIT;

-- Si hay error
-- ROLLBACK;

-- Procedimiento almacenado
CREATE PROCEDURE AumentarSalarios(IN porcentaje DECIMAL(5,2))
BEGIN
    UPDATE Empleados 
    SET salario = salario * (1 + porcentaje/100);
END;

-- Llamada al procedimiento
CALL AumentarSalarios(5.0);
